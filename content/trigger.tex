\begin{frame}{The Situation}
    \begin{itemize}
        \item HLT in one process
        \item HLT2 parts rely on properties computed in HLT1
        \item[]
        \item[$\Rightarrow$] Need to separate HLT1,2
    \end{itemize}
\end{frame}

\begin{frame}{Getting There}
    \begin{itemize}
        \item Important: Find dependecies
        \item ``Scalpel" approach \begin{enumerate}
            \item Remove HLT1 lines
            \item From errors find dependencies
            \item Resolve until there are none
        \end{enumerate}
        \item Good insight into the Trigger
    \end{itemize}
\end{frame}

\begin{frame}{For Example}
    \begin{itemize}
        \item PV3D \begin{itemize}
            \item Used to be in HLT1
            \item Couldn't be instantiated from HLT2
            \item[$\Rightarrow$] Bug in LoKi
        \end{itemize}
        \item TISTOS decisions \begin{itemize}
            \item Only produced in HLT1
            \item[$\Rightarrow$] Remove every single occurence in the HLT2 lines
        \end{itemize}
    \end{itemize}
\end{frame}
